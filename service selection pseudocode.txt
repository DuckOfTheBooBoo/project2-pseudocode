procedure serviceSelection(output character cServiceChoice)
begin
	Display "Select Service: ";
	Display "1. Plumber";
	Display "2. Electrician";
	Display "3. Carpenter";
	Display "4. Air Conditioner Technician";
	
	call inputValidation(
		"Select service: ", 
		["1", "2", "3", "4"], 
		"Invalid input. Please enter number from 1 to 4.",
		cServiceChoice
	);
	
	switch (cServiceChoice)
		begin
			case "1":
				call Plumber();
				break;
			case "2":
				call Electrician();
				break;
			case "3":
				call Carpenter();
				break;
			case "4":
				call AirConditionerTechnician();
				break;
			end
	endif
end
