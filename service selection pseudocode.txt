procedure serviceSelection
begin
	Character cServiceChoice;
	Display "Select Service: ";
	Display "1. Plumber";
	Display "2. Electrician";
	Display "3. Carpenter";
	Display "4. Septic Tank Cleaner";
	
	nServiceChoice = call inputValidation(
		cPrompt = "Select service: ", 
		arrValidInput = ["1", "2", "3", "4", "5"], 
		cErrorMsg = "Invalid input. Please enter number from 1 to 5."
	);
	
	switch (cServiceChoice)
		begin
			case "1":
				Call plumberService;
				break;
			case "2":
				Call electricianService;
				break;
			case "3":
				Call carpenterService;
				break;
			case "4":
				Call septicTank;
				break;
			end
	endif
end
