procedure Plumber(output character cTitle, output character cDescription, output date dDate, output character cReserveDuration, output numeric nServiceCost)
begin
    Character cProblemChoice;
    Display "What can we do for you?";
    Display "1. Install cabinets";
    Display "2. Repair damaged woods";
    Display "3. Repair damaged doors";
    Display "4. Repair wooden furnitures";
    Display "5. Other";
    
    call inputValidation(
        "Select problem: ",
        ["1", "2", "3", "4", "5"],
        "Invalid input. Please enter number from 1 to 5.",
        cProblemChoice
    );

    switch (cProblemChoice):
        begin
            case "1":
                cTitle = "Install cabinets";
                break;
            
            case "2":
                cTitle = "Repair damaged woods";
                break;

            case "3":
                cTitle = "Repair damaged doors";
                break;

            case "4":
                cTitle = "Repair wooden furnitures";

            case "5":
                Display "Custom title: ";
                Accept cTitle;
                break;
        end
    
    Display "Description: ";
    Accept cDescription;
    call DateSelection(dDate);
    call ReserveDuration(cReserveDuration, nServiceCost);
    
end
